10  class AirQualityTest {
11      @Test
12      void givenForecastsCategoryAboveThreshold_whenCheckingAlarm_thenAlarmIsTriggered() {
13          Forecast[] forecasts = {
14                  new Forecast()
15                          .withCategory(new Category().withNumber(3))
16                          .withParameterName("O3")
17                          .withActionDay(true)
18          };
19  
20          AirQuality airQuality = new AirQuality();
21          AQI[] aqis = airQuality.alarmTriggered(forecasts);
22          assertTrue(aqis.length > 0, "should be triggered when category is 3 or higher");
23      }
24  
25      @Test
26      void givenForecastCategoryBelowThreshold_whenCheckingAlarm_thenAlarmIsTriggered() {
27          Forecast[] forecasts = {
28                  new Forecast()
29                          .withCategory(new Category().withNumber(1))
30                          .withParameterName("O3")
31                          .withActionDay(false)
32          };
33  
34          AirQuality airQuality = new AirQuality();
35          AQI[] aqis = airQuality.alarmTriggered(forecasts);
36          assertTrue(aqis.length == 0, "should not be triggered when category is below 3");
37      }
