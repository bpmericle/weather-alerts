27  @ExtendWith(MockitoExtension.class)
28  class AirNowTest {
29      private static final int ZIP = 76226;
30      private static final String FORECASTS_BY_ZIP_CODE = "airnow_forecasts_by_zip_code.json";
31      private static final String OBSERVATIONS_BY_ZIP_CODE = "airnow_observations_by_zip_code.json";
32  
33      @Mock
34      private HttpClient mockHttpClient;
35  
36      @Mock
37      private HttpResponse<String> mockResponse;
38  
39      @BeforeEach
40      void beforeEach() {
41          reset(mockHttpClient, mockResponse);
42      }
43  
44      @Test
45      void givenOfflineMode_whenGetForecastsByZipCodeCalled_thenDataIsReturned() {
46          try {
47              String contents = DataLoader.loadFromFile(FORECASTS_BY_ZIP_CODE);
48              when(mockResponse.statusCode()).thenReturn(HttpURLConnection.HTTP_OK);
49              when(mockResponse.body()).thenReturn(contents);
50  
51              when(mockHttpClient.send(any(HttpRequest.class), ArgumentMatchers.<BodyHandler<String>>any()))
52                      .thenReturn(mockResponse);
53  
54              AirNow airNow = new AirNow(mockHttpClient);
55              Forecast[] forecasts = airNow.getForecastByZipCode(ZIP);
56  
57              verify(mockResponse).statusCode();
58              verify(mockResponse).body();
59              verify(mockHttpClient).send(any(HttpRequest.class), ArgumentMatchers.<BodyHandler<String>>any());
60  
61              assertNotNull(forecasts, "forecasts should not be null");
62              assertTrue(forecasts.length > 0, "should contain forecasts");
63          } catch (Exception e) {
64              fail("No exception is expected when running this test", e);
65          }
66      }
67  
68      @Test
69      void givenOfflineMode_whenGetObservationsByZipCodeCalled_thenDataIsReturned() {
