3   import javax.mail.MessagingException;
4   
5   import com.mericle.weather.airnow.alarm.AirQuality;
6   import com.mericle.weather.airnow.model.AQI;
7   import com.mericle.weather.airnow.model.Forecast;
8   import com.mericle.weather.airnow.service.AirNow;
9   import com.mericle.weather.util.Email;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  public class App {
14      private static final Logger LOGGER = LoggerFactory.getLogger(App.class);
15      private static final String SUBJECT_PATTERN = "Weather Alarm [%s] Has Been Triggered!";
16      private static final String CONTENT_TEMPLATE = "<html>" +
17              "    <head>" +
18              "      <style>" +
19              "        .key {" +
20              "          font-weight: bold;" +
21              "        }" +
22              "        #body {" +
23              "          font-size: 16px;" +
24              "          width: 794px;" +
25              "          margin-bottom: 10px;" +
26              "        }" +
27              "      </style>" +
28              "    </head>" +
29              "    <body>" +
30              "      <div id='body'>" +
31              "        %s" +
32              "      </div>" +
33              "      <img src=\"https://www.epa.gov/sites/default/files/2014-09/aqiguidepm.png\" alt=\"AQI Guide\" />" +
34              "      <div>Click <a href=\"https://www.airnow.gov/aqi/aqi-basics/\">here for more information</a>.</div>" +
35              "    </body>" +
36              "</html>";
37  
38      public static void main(String[] args) {
39          Forecast[] forecasts = new AirNow().getForecastByZipCode(76226);
40  
41          AirQuality alarm = new AirQuality();
42          AQI[] aqis = alarm.alarmTriggered(forecasts);
43          LOGGER.info("forecasts={}\r\n\r\naqis={}", forecasts, aqis);
44  
45          if (aqis.length > 0) {
46              LOGGER.info("Alarm has been triggered");
47  
48              String subject = String.format(SUBJECT_PATTERN, "Air Quality");
49              String content = buildBody(aqis);
50              LOGGER.info("sending email, subject={}, body={}", subject, content);
51  
52              try {
53                  Email email = new Email();
54                  email.send(subject, content);
55              } catch (MessagingException e) {
56                  LOGGER.error("Unable to send email.", e);
57              }
58          }
59      }
60  
61      private static String buildBody(AQI[] aqis) {
62          StringBuilder builder = new StringBuilder();
63  
64          for (AQI aqi : aqis) {
65              builder.append(
66                      String.format("<div style=\"background-color:#%s;color:white;padding:10px;margin-bottom:5px;\">",
67                              aqi.hexColor))
68                      // .append("<p><span class='key'>Air Quality Index (AQI):</span> ")
69                      // .append(aqi.index)
70                      // .append("</p>")
71                      .append("<p><span class='key'>Level of Health Concern:</span> ")
72                      .append(aqi.description)
73                      .append("</p>")
74                      .append("<p><span class='key'>Triggered By:</span> ")
75                      .append(aqi.parameterName)
76                      .append("</p>")
77                      .append("</div>");
78          }
79          return String.format(CONTENT_TEMPLATE, builder.toString());
80      }
81  }
