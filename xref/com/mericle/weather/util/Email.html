1   package com.mericle.weather.util;
2   
3   import java.util.Objects;
4   import java.util.Properties;
5   
6   import javax.mail.Message;
7   import javax.mail.MessagingException;
8   import javax.mail.Multipart;
9   import javax.mail.PasswordAuthentication;
10  import javax.mail.Session;
11  import javax.mail.Transport;
12  import javax.mail.internet.InternetAddress;
13  import javax.mail.internet.MimeBodyPart;
14  import javax.mail.internet.MimeMessage;
15  import javax.mail.internet.MimeMultipart;
16  
17  public class Email {
18      private static final String FROM_EMAIL_KEY = "from";
19      private static final String TO_EMAIL_KEY = "to";
20      private static final String SMTP_PASSWORD_KEY = "smtp.password";
21      private static final String DEFAULT_FROM = "brian.mericle@gmail.com";
22  
23      public void send(String subject, String content) throws MessagingException {
24          String from = System.getProperty(FROM_EMAIL_KEY, DEFAULT_FROM);
25          String password = System.getProperty(SMTP_PASSWORD_KEY);
26          Objects.requireNonNull(password,
27                  String.format("%s is required to be set on the command line. (i.e. -D%s=<value>)", SMTP_PASSWORD_KEY,
28                          SMTP_PASSWORD_KEY));
29  
30          // emily.mericle@argyleisd.com
31          String to = System.getProperty(TO_EMAIL_KEY, DEFAULT_FROM);
32  
33          Properties properties = new Properties();
34          properties.put("mail.smtp.user", from);
35          properties.put("mail.smtp.host", "smtp.gmail.com");
36          properties.put("mail.smtp.port", "465");
37          properties.put("mail.smtp.auth", "true");
38          properties.put("mail.smtp.socketFactory.port", "465");
39          properties.put("mail.smtp.socketFactory.class", "javax.net.ssl.SSLSocketFactory");
40          properties.put("mail.smtp.socketFactory.fallback", "false");
41          properties.put("mail.smtp.starttls.enable", "true");
42  
43          // Get the Session object.// and pass username and password
44          Session session = Session.getInstance(properties, new javax.mail.Authenticator() {
45              protected PasswordAuthentication getPasswordAuthentication() {
46                  return new PasswordAuthentication(from, password);
47              }
48          });
49  
50          session.setDebug(true);
51  
52          MimeMessage message = new MimeMessage(session);
53          message.setFrom(new InternetAddress(from));
54          message.addRecipient(Message.RecipientType.TO, new InternetAddress(to));
55          message.setSubject(subject);
56  
57          MimeBodyPart htmlPart = new MimeBodyPart();
58          htmlPart.setContent(content, "text/html; charset=utf-8");
59  
60          Multipart multiPart = new MimeMultipart(htmlPart);
61          message.setContent(multiPart);
62  
63          Transport.send(message);
64      }
65  }
