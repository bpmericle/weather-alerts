1   package com.mericle.weather.airnow.alarm;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import com.mericle.weather.airnow.model.AQI;
7   import com.mericle.weather.airnow.model.Forecast;
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  
11  public class AirQuality {
12      private static final Logger LOGGER = LoggerFactory.getLogger(AirQuality.class);
13      private static final int ALARM_THRESHOLD = 3;
14  
15      public AQI[] alarmTriggered(Forecast[] forecast) {
16          List<AQI> triggered = new ArrayList<>();
17          for (Forecast f : forecast) {
18              AQI aqi = AQI.byCategory(f.getCategory().getNumber());
19              if (aqi.category >= ALARM_THRESHOLD) {
20                  aqi.index = f.getAqi();
21                  aqi.parameterName = f.getParameterName();
22                  aqi.actionDay = f.isActionDay();
23  
24                  LOGGER.debug("air quality triggered on {}", aqi);
25                  triggered.add(aqi);
26              }
27          }
28  
29          return triggered.toArray(new AQI[0]);
30      }
31  }
